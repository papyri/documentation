def processargs
  if ARGV.length >= 1 && ARGV[0] == '--help'
    puts "Usage: #{$0} < input.#{@from} > output.#{@to}\n" +
         "       #{$0} grammar.xsg < input.#{@from} > output.#{@to}\n" +
         "       #{$0} grammar.xsg input.#{@from} > output.#{@to}"
    Process.exit
  end

  if ARGV.length >= 1
    @grammar_file = ARGV[0]
    if ARGV.length >= 2
      @from_file = ARGV[1]
    end
  end
end
